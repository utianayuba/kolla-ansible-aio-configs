#1. Download Fedora-CoreOS image and upload to OpenStack glance image service
source karno-openrc.sh 
wget -c https://builds.coreos.fedoraproject.org/prod/streams/stable/builds/35.20220213.3.0/x86_64/fedora-coreos-35.20220213.3.0-openstack.x86_64.qcow2.xz
xz -d fedora-coreos-35.20220213.3.0-openstack.x86_64.qcow2.xz
openstack image create --disk-format qcow2 --file fedora-coreos-35.20220213.3.0-openstack.x86_64.qcow2 --property os_distro='fedora-coreos' --progress fedora-coreos-35.20220213.3.0-openstack.x86_64
openstack image list

#2. Create a cluster template for a Kubernetes
openstack coe cluster template create template-kubernetes-v1.21.9-rancher1 --coe kubernetes --image fedora-coreos-35.20220213.3.0-openstack.x86_64 --external-network net-ext --master-flavor flavor-4-4-40 --flavor flavor-4-4-40 --network-driver calico --dns-nameserver 10.11.11.11 --labels container_infra_prefix=registry.stratus.ok:4433/,hyperkube_prefix=registry.stratus.ok:4433/,kube_tag=v1.21.9-rancher1,cloud_provider_tag=v1.21.0,heat_container_agent_tag=wallaby-stable-1,cinder_csi_plugin_tag=v1.21.0,k8s_keystone_auth_tag=v1.21.0,magnum_auto_healer_tag=v1.21.1,autoscaler_tag=v1.21.0,octavia_ingress_controller_tag=v1.21.0,boot_volume_size=40,ingress_controller=octavia,cloud_provider_enabled=true,volume_driver=cinder,cinder_csi_enabled=true,csi_attacher_tag=v2.2.1,csi_provisioner_tag=v1.6.1,csi_snapshotter_tag=v2.1.3,csi_resizer_tag=v0.5.1,csi_node_driver_registrar_tag=v1.3.0,auto_healing_enabled=true,auto_healing_controller=magnum-auto-healer
openstack coe cluster template list

#3. Create a cluster
openstack coe cluster create kubernetes-v1.21.9-rancher1 --cluster-template template-kubernetes-v1.21.9-rancher1 --keypair key-0 --master-count 1 --master-flavor flavor-4-4-40 --node-count 1 --flavor flavor-4-4-40
openstack coe cluster list

#4. Check cluster
mkdir ~/bin
wget -c https://dl.k8s.io/release/v1.21.9/bin/linux/amd64/kubectl -O ~/bin/kubectl
chmod +x ~/bin/kubectl 
mkdir -p ~/clusters/kubernetes-v1.21.9-rancher1
rm -rf ~/clusters/kubernetes-v1.21.9-rancher1/config
$(openstack coe cluster config kubernetes-v1.21.9-rancher1 --dir ~/clusters/kubernetes-v1.21.9-rancher1)
export KUBECONFIG=~/clusters/kubernetes-v1.21.9-rancher1/config
kubectl -n kube-system get po
